Line Follower Robot

Reads two IR sensors and adjusts motor speeds to follow a black line on the ground.

Sensor fusion + motor control
line_follower.kx
program {
    loop forever {
        make var left  = read analog pin 0
        make var right = read analog pin 1

        if left > 500 and right > 500 {
            set pin 9 to 200
            set pin 10 to 200
        } else if left < 500 {
            set pin 9 to 200
            set pin 10 to 50
        } else {
            set pin 9 to 50
            set pin 10 to 200
        }
        wait 10
    }
}

Function-Based Robot

Reusable motor functions ‚Äî forward, turn left, turn right, stop. Clean and modular robot code.

Functions + parameters
robot.kx
program {
    def forward(speed) {
        set pin 9 to speed
        set pin 10 to speed
    }
    def turn_left(speed) {
        set pin 9 to 0
        set pin 10 to speed
    }
    def stop_robot() {
        set pin 9 to 0
        set pin 10 to 0
    }
    loop forever {
        forward(200)
        wait 2000
        turn_left(150)
        wait 500
        stop_robot()
        wait 1000
    }
}

4-Bit Binary Counter

Counts 0‚Äì15 in binary, showing each bit on a separate LED. Uses modulo arithmetic and bit-shifting logic.

Advanced math + GPIO
binary_counter.kx
program {
    make var count = 0
    loop forever {
        if count % 2 == 1 {
            turn on pin 10
        } else { turn off pin 10 }
        if (count / 2) % 2 == 1 {
            turn on pin 11
        } else { turn off pin 11 }
        if (count / 4) % 2 == 1 {
            turn on pin 12
        } else { turn off pin 12 }
        change count by 1
        if count >= 16 {
            change count by -16
        }
        wait 500
    }
}

Self-Balancing Robot

PID-style balance control using analog sensors and PWM motor output. Advanced real-time robotics.

Real-time control + math
balance.kx
program {
    const KP = 2
    const SETPOINT = 512

    loop forever {
        make var tilt = read analog pin 0
        make var error = SETPOINT - tilt
        make var output = abs(error) * KP

        make var speed = constrain(output, 0, 255)

        if error > 0 {
            set pin 9 to speed
            set pin 10 to 0
        } else {
            set pin 9 to 0
            set pin 10 to speed
        }
        wait 10
    }
}

Get Started

Up and Running in 5 Minutes

01

Clone the Repo

git clone https://github.com/sohammulik/kinetrix
cd kinetrix
02

Build the Compiler

gcc -o kcc compiler_v3.c parser.c \
    codegen.c ast.c error.c \
    pin_tracker.c symbol_table.c \
    diagnostics.c -lm
03

Write & Compile

# Write your program
nano my_robot.kx

# Compile it
./kcc my_robot.kx -o my_robot.ino
