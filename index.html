Kinetrix is a compiled, statically-typed language for embedded systems. No boilerplate, no header hell — just hardware control that reads like your thoughts.

balance_robot.kx
include "sensors/mpu6050.kx"
include "control/pid.kx"

shared make int emergency = 0

task safety_monitor {
    loop forever {
        make var temp = read analog pin 1
        if temp > 800 { set emergency to 1 }
        wait 100
    }
}

program {
    mpu6050_init()
    pid_init(2.0, 0.1, 0.05)
    pid_set_limits(-255, 255)
    start task safety_monitor
    enable watchdog timeout 2000ms

    loop forever {
        feed watchdog
        mpu6050_read_accel()
        make var output = pid_compute(512, accel_x, 0.01)
        if emergency is 0 {
            set pin 9 to constrain(output, 0, 255)
        }
        wait 10
    }
}
compiles to
balance_robot.ino (generated)
// ⚡ Generated by Kinetrix v3.1
#include <Arduino.h>
#include <Wire.h>
#include <esp_task_wdt.h>

volatile int emergency = 0;

void TaskSafetyMonitor(void* pvParams) {
  for(;;) {
    int temp = analogRead(1);
    if(temp > 800) emergency = 1;
    vTaskDelay(100 / portTICK_PERIOD_MS);
  }
}

void setup() {
  Wire.begin();
  esp_task_wdt_init(2, true);
  esp_task_wdt_add(NULL);
  mpu6050_init();
  pid_init(2.0, 0.1, 0.05);
  pid_set_limits(-255, 255);
  xTaskCreate(TaskSafetyMonitor,
    "safety", 4096, NULL, 1, NULL);
}

void loop() {
  esp_task_wdt_reset();
  mpu6050_read_accel();
  float out = pid_compute(512,accel_x,0.01);
  if(emergency == 0)
    ledcWrite(0, constrain(out,0,255));
  delay(10);
}

Tasks & Concurrency

Hardware Interrupts

Pin and timer interrupts with IRAM-safe ISR generation. Encoder counting, edge detection, and precise timers.

on pin 2 rising {
    change pulse_count by 1
}
on timer every 500 ms {
    feed watchdog
}

🛡️

Watchdog & Safety

Hardware watchdog reboots if your robot hangs. Shared volatile variables, interrupt guards, and try/on error blocks.

enable watchdog timeout 2000ms
loop forever {
    feed watchdog
    try { # I2C read }
    on error { print "fail" }
}
NEW v3.1
🧱

Encoder RPM Counter

Uses a hardware interrupt on pin 2 to count encoder pulses. Safely reads the counter using interrupt guards to prevent race conditions.

Hardware interrupts + shared vars
encoder.kx
shared make int pulse_count = 0

on pin 2 rising {
    change pulse_count by 1
}

program {
    loop forever {
        disable interrupts
        make int count = pulse_count
        set pulse_count to 0
        enable interrupts

        make float rpm = count * 60.0
        println "RPM: " + rpm
        wait 1 seconds
    }
}

Multi-task Safety System

Two FreeRTOS tasks run in parallel: a safety monitor watching temperature, and a motor controller that respects the emergency stop flag.

FreeRTOS tasks + shared variables
multitask.kx
shared make int emergency = 0

task safety_monitor {
    loop forever {
        make var temp = read analog pin 1
        if temp > 800 {
            set emergency to 1
            println "Overheat!"
        }
        wait 100
    }
}

task motor_control {
    loop forever {
        if emergency is 1 {
            set pin 9 to 0
            println "EMERGENCY STOP"
        } else {
            set pin 9 to 180
        }
        wait 50
    }
}

program {
    start task safety_monitor
    start task motor_control
    enable watchdog timeout 5000ms
    loop forever { feed watchdog  wait 500 }
}

PID Motor Controller

Uses the stdlib PID module and encoder interrupts to drive a motor at a precise target speed, automatically correcting for load changes.

stdlib PID + hardware interrupt
pid_motor.kx
include "control/pid.kx"

const float TARGET = 200.0
const int   ENCODER_PIN = 2
const int   MOTOR_PIN = 9

shared make int tick_count = 0

on pin ENCODER_PIN rising {
    change tick_count by 1
}

program {
    pid_init(2.0, 0.1, 0.05)
    pid_set_limits(0, 255)

    loop forever {
        disable interrupts
        make int ticks = tick_count
        set tick_count to 0
        enable interrupts

        make float speed = ticks * 10.0
        make float out = pid_compute(TARGET, speed, 0.1)
        set pin MOTOR_PIN to cast int out
        wait 100
    }
}

Wireless Telemetry

A sensor task reads data and broadcasts it via ESP-NOW radio to a base station, while also receiving motor commands wirelessly.

ESP-NOW + multi-task + watchdog
telemetry.kx
shared make float sensor_value = 0.0
shared make bool  new_data = false

task sensor_reader {
    loop forever {
        set sensor_value to read analog pin 0
        set new_data to true
        wait 200
    }
}

program {
    start task sensor_reader
    enable watchdog timeout 5000ms
    make int base_station = 0

    loop forever {
        feed watchdog
        if new_data {
            radio_send_peer(base_station, sensor_value)
            set new_data to false
        }
        if radio_available() {
            make float cmd = radio_read()
            set pin 9 to cast int cmd
        }
        wait 50
    }
}

Full Navigation with Kalman Filter

Production-grade navigation: MPU6050 IMU, GPS, Compass, and a 2D Kalman filter for position estimation. The complete robotics stack in Kinetrix.

IMU + GPS + Compass + Kalman 2D
navigation.kx
include "sensors/mpu6050.kx"
include "sensors/neo6m_gps.kx"
include "sensors/hmc5883l_compass.kx"
include "fusion/kalman_2d.kx"
include "fusion/calibration.kx"

program {
    mpu6050_init()
    gps_init()
    hmc5883l_init()
    calibrate_imu_accel(100)
    calibrate_imu_gyro(100)
    calibrate_compass(200)
    set_magnetic_declination(5)
    kalman_2d_init(0, 0, 0, 0)

    loop forever {
        mpu6050_read_accel()
        hmc5883l_read()
        gps_process_byte()

        make var ax = accel_x - imu_accel_offset_x
        make var ay = accel_y - imu_accel_offset_y

        kalman_2d_predict(0.1)
        kalman_2d_update_accel(ax, ay, 0.1)

        if has_gps_fix() {
            kalman_2d_update(get_latitude(), get_longitude())
        }

        make var x = kalman_2d_get_x()
        make var y = kalman_2d_get_y()
        make var hdg = apply_compass_calibration()
        make var vx = kalman_2d_get_vx()
        make var vy = kalman_2d_get_vy()
        make float spd = sqrt(vx*vx + vy*vy)

        print "X: " + x + " Y: " + y
        println " Hdg: " + hdg + " Spd: " + spd
        wait 100
    }
}

Up and Running in 5 Minutes

⚡ One-line install (Mac / Linux)
curl -fsSL https://raw.githubusercontent.com/M4dM0nk3y/Kinetrix/main/install.sh | bash
or install manually
01

Clone the Repo

git clone https://github.com/
  M4dM0nk3y/Kinetrix
cd Kinetrix
02

Build the Compiler

gcc -o kcc compiler_v3.c \
  parser.c codegen.c \
  ast.c error.c \
  pin_tracker.c \
  symbol_table.c -lm
03

Write & Compile

# Default: ESP32 target
./kcc robot.kx

# Other targets:
./kcc robot.kx --target pico
./kcc robot.kx --target rpi
./kcc robot.kx --target ros2
04
